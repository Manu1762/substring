#include <iostream>
using namespace std;

int main()
{
    int rows;
   int k;
    cout << "Enter number of rows: ";
    cin >> rows;
    k=0;
    for(int i = rows; i >= 1; --i)
    {
        for(int p=k;p>0;--p){
        cout<<" ";
        }
         k = k + 1;
        for(int j = k; j <= rows; ++j)
        {
            cout << j << " ";
        }
        cout << endl;
    }
for(int i = rows; i > 1; --i)
    {
        for(int p=k-1;p>1;--p){
        cout<<" ";
        }
         k = k - 1;
        for(int j = k; j <= rows; ++j)
        {
            cout << j << " ";
        }
        cout << endl;
    }

    return 0;
}
